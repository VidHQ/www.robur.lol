Module common.cave

This module offers functions to intercept functions with codecaves

Examples

-- simple hook you can access arguments with context:getArg(index, "cdef")
-- its also possible to change arguments on the stack with context:setArg(1, 1234)
function hook_function(context)
    local arg1 = context:getArg(1, "uint32_t")
    local arg2 = context:getArg(2, "wchar_t*")
    local eax = context.reg32.eax
    local ecx = context.reg32.ecx

    context.reg32.eax = 0x1337
    context:setArg(1, 0x1337)
end
cave.intercept_pre("PreCave", 0x123456, hook_function)
